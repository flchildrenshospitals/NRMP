didDrawCell: (data) => { // Only do this for body rows and only once per row: if (data.row.section === "body" && data.row.raw.isMedianBorder && data.column.index === 0) { const doc = data.doc; // Find the left edge and right edge of this row let xStart = Infinity; let xEnd = -Infinity; // Go through all cells in the row Object.keys(data.row.cells).forEach((key) => { const cell = data.row.cells[key]; const cellLeft = cell.x; const cellRight = cell.x + cell.width; if (cellLeft < xStart) xStart = cellLeft; if (cellRight > xEnd) xEnd = cellRight; }); // The dashed line is typically at the *bottom* of the row const dashY = data.cell.y; // Draw the dashed line doc.setLineDash([3, 3], 0); doc.setDrawColor(0, 51, 102); doc.setLineWidth(0.5); doc.line(xStart, dashY, xEnd, dashY); // Reset dash doc.setLineDash([]); doc.setLineWidth(0.2); } }
